`webapp/frontend/lib/api.ts`:

```typescript
import type { AssessRequest, AssessResponse, AssessRow, DownloadRequest } from "./types";

const BASE = process.env.NEXT_PUBLIC_API_BASE_URL ?? "http://localhost:8000";

async function handleResponse<T>(res: Response): Promise<T> {
  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    const msg = body?.detail?.message ?? body?.error?.message ?? res.statusText;
    throw new Error(msg);
  }
  return res.json() as Promise<T>;
}

export async function assess(req: AssessRequest): Promise<AssessResponse> {
  const res = await fetch(`${BASE}/api/assess`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(req),
  });
  return handleResponse<AssessResponse>(res);
}

export async function downloadXlsx(
  request: AssessRequest,
  rows: AssessRow[]
): Promise<void> {
  const body: DownloadRequest = { request, rows };
  const res = await fetch(`${BASE}/api/download`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  if (!res.ok) {
    const b = await res.json().catch(() => ({}));
    throw new Error(b?.detail?.message ?? "다운로드 실패");
  }
  const blob = await res.blob();
  const disposition = res.headers.get("content-disposition") ?? "";
  const match = disposition.match(/filename="?([^"]+)"?/);
  const filename = match?.[1] ?? "위험성평가서.xlsx";
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}
```

```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table";
import type { AssessRow, AssessMeta } from "@/lib/types";

interface Props {
  rows: AssessRow[];
  meta: AssessMeta;
  onChange: (rows: AssessRow[]) => void;
  onDownload: () => void;
  onRegenerate: () => void;
  isDownloading: boolean;
}

export default function ResultTable({
  rows,
  meta,
  onChange,
  onDownload,
  onRegenerate,
  isDownloading,
}: Props) {
  function updateCell(
    idx: number,
    field: keyof AssessRow,
    value: string
  ) {
    const updated = rows.map((r, i) =>
      i === idx ? { ...r, [field]: value } : r
    );
    onChange(updated);
  }

  function addRow() {
    onChange([
      ...rows,
      { location: "", work: "", hazard: "", control: "", note: "" },
    ]);
  }

  function removeRow(idx: number) {
    onChange(rows.filter((_, i) => i !== idx));
  }

  return (
    <div className="flex flex-col gap-3">
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <span className="text-sm font-medium">생성 결과 ({rows.length}행)</span>
          {meta.fallback_used && (
            <Badge variant="secondary">AI 보강 실패 — 검색 원문 표시</Badge>
          )}
          <Badge variant="outline" className="text-xs">
            {meta.model}
          </Badge>
        </div>
        <div className="flex gap-2">
          <Button variant="outline" size="sm" onClick={onRegenerate}>
            다시 생성
          </Button>
          <Button size="sm" onClick={onDownload} disabled={isDownloading}>
            {isDownloading ? "다운로드 중..." : "엑셀 다운로드"}
          </Button>
        </div>
      </div>

      <div className="overflow-x-auto rounded border">
        <Table>
          <TableHeader>
            <TableRow>
              <TableHead className="w-8">#</TableHead>
              <TableHead className="w-24">작업장소</TableHead>
              <TableHead className="w-24">작업내용</TableHead>
              <TableHead>위험요인</TableHead>
              <TableHead>안전보건추진계획</TableHead>
              <TableHead className="w-16">비고</TableHead>
              <TableHead className="w-10"></TableHead>
            </TableRow>
          </TableHeader>
          <TableBody>
            {rows.map((row, idx) => (
              <TableRow key={idx}>
                <TableCell className="text-center text-sm text-muted-foreground">
                  {idx + 1}
                </TableCell>
                <TableCell>
                  <Input
                    value={row.location}
                    onChange={(e) => updateCell(idx, "location", e.target.value)}
                    className="h-8 text-sm"
                  />
                </TableCell>
                <TableCell>
                  <Input
                    value={row.work}
                    onChange={(e) => updateCell(idx, "work", e.target.value)}
                    className="h-8 text-sm"
                  />
                </TableCell>
                <TableCell>
                  <Textarea
                    value={row.hazard}
                    onChange={(e) => updateCell(idx, "hazard", e.target.value)}
                    rows={3}
                    className="text-sm"
                  />
                </TableCell>
                <TableCell>
                  <Textarea
                    value={row.control}
                    onChange={(e) => updateCell(idx, "control", e.target.value)}
                    rows={3}
                    className="text-sm"
                  />
                </TableCell>
                <TableCell>
                  <Input
                    value={row.note}
                    onChange={(e) => updateCell(idx, "note", e.target.value)}
                    className="h-8 text-sm"
                  />
                </TableCell>
                <TableCell>
                  <Button
                    variant="ghost"
                    size="sm"
                    onClick={() => removeRow(idx)}
                    className="h-7 w-7 p-0 text-destructive"

                    ✕

}
```

- [ ] **Step 3: page.tsx 작성**

`webapp/frontend/app/page.tsx`:
```tsx
"use client";

import { useState } from "react";
import AssessForm from "@/components/AssessForm";
import ResultTable from "@/components/ResultTable";
import { assess, downloadXlsx } from "@/lib/api";
import type { AssessRequest, AssessResponse } from "@/lib/types";

export default function HomePage() {
  const [isLoading, setIsLoading] = useState(false);
  const [isDownloading, setIsDownloading] = useState(false);
  const [result, setResult] = useState<AssessResponse | null>(null);
  const [currentRequest, setCurrentRequest] = useState<AssessRequest | null>(null);
  const [error, setError] = useState<string | null>(null);

  async function handleSubmit(req: AssessRequest) {
    setIsLoading(true);
    setError(null);
    setCurrentRequest(req);
    try {
      const res = await assess(req);
      setResult(res);
    } catch (e) {
      setError(e instanceof Error ? e.message : "알 수 없는 오류");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleDownload() {
    if (!result || !currentRequest) return;
    setIsDownloading(true);
    try {
      await downloadXlsx(currentRequest, result.rows);
    } catch (e) {
      setError(e instanceof Error ? e.message : "다운로드 실패");
    } finally {
      setIsDownloading(false);
    }
  }

  function handleRegenerate() {
    if (currentRequest) handleSubmit(currentRequest);
  }

  return (
    <div className="min-h-screen bg-slate-50">
      <header className="border-b bg-white px-6 py-4">
        <h1 className="text-xl font-bold text-slate-800">
          AI 위험성평가 도우미
          <span className="ml-2 text-sm font-normal text-slate-500">
            불시·단발성 작업
          </span>
        </h1>
      </header>

      <main className="mx-auto max-w-screen-xl px-6 py-6">
        {error && (
          <div className="mb-4 rounded bg-red-50 border border-red-200 px-4 py-3 text-sm text-red-700">
            {error}
          </div>
        )}

        <div className="flex gap-6">
          {/* 좌측: 입력 폼 */}
          <div className="w-80 flex-shrink-0">
            <AssessForm onSubmit={handleSubmit} isLoading={isLoading} />
          </div>

          {/* 우측: 결과 */}
          <div className="flex-1">
            {isLoading && (
              <div className="flex h-64 items-center justify-center rounded border bg-white">
                <div className="text-center">
                  <div className="mb-2 h-8 w-8 animate-spin rounded-full border-4 border-primary border-t-transparent mx-auto" />
                  <p className="text-sm text-muted-foreground">
                    AI가 위험요인을 분석 중입니다...
                  </p>
                </div>
              </div>
            )}

            {!isLoading && result && (
              <div className="rounded border bg-white p-4">
                <ResultTable
                  rows={result.rows}
                  meta={result.meta}
                  onChange={(rows) => setResult({ ...result, rows })}
                  onDownload={handleDownload}
                  onRegenerate={handleRegenerate}
                  isDownloading={isDownloading}
                />
              </div>
            )}

            {!isLoading && !result && (
              <div className="flex h-64 items-center justify-center rounded border border-dashed bg-white text-sm text-muted-foreground">
                왼쪽 폼을 작성하고 [AI 생성] 버튼을 클릭하세요
              </div>
            )}
          </div>
        </div>
      </main>
    </div>
  );
}
```
